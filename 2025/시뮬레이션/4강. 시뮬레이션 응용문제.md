```c
void poissn(long *np, float mean, int *pp)
{
  float prod, b, u;
  *pp=0;
  b = exp(-mean);
  prod=1;
  random(np, &u);
  
  prod = prod*u;
  
  while (prod >= b)
  {
    random(np, &u);
    prod = prod*u;
    ++(*pp);
  }
}
```
